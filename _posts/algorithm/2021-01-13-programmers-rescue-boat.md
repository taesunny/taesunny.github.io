# Solution

``` java
import java.util.Arrays;

class Solution {
    public int solution(int[] people, int limit) {
        int answer = 0;
        
        Arrays.sort(people);
        int minValueIndex = 0;
        
        for(int maxValueIndex=people.length-1; maxValueIndex>=minValueIndex; maxValueIndex--) {
            
            if(people[maxValueIndex] + people[minValueIndex] <= limit) {
                minValueIndex++;
            }
            
            answer++;
        }

        return answer;
    }
}
```
